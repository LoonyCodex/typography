## H2

```tsx
import React from 'react';
import { H2 } from 'typography';

export default () => <H2>H2 with default styles</H2>;
```

### Size

```tsx
import React from 'react';
import { H2 } from 'typography';

export default () => (<>
	<H2 fontSize="large">H2 large</H2>
	<H2 fontSize="large">H2 large</H2>
	<H2 fontSize="xlarge">H2 xlarge</H2>
	<H2 fontSize="xxlarge">H2 xxlarge</H2>
</>);
```

### Color

import React from 'react';
